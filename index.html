function setupRealtimeListeners() { const q = query(collection(window.db, "reservations")); let isFirstLoad = true; // ⭐ 前回の状態を記憶するための臨時の箱 let previousStatuses = {}; onSnapshot(q, (snapshot) => { let data = []; let triggerChime = false; snapshot.forEach((doc) => { const currentData = doc.data(); const id = doc.id; data.push({ id: id, ...currentData }); // 最初の読み込み（画面を開いた瞬間）は音を鳴らさない if (!isFirstLoad) { const oldStatus = previousStatuses[id]; const newStatus = currentData.status; // ⭐【ここが重要】「前の状態が対応中ではなく」かつ「新しい状態が対応中になった」ときだけフラグをONにする if (oldStatus !== "対応中" && newStatus === "対応中") { triggerChime = true; } } // 最新のステータスを記憶しておく previousStatuses[id] = currentData.status; }); data.sort((a, b) => a.time.localeCompare(b.time)); localReservations = data; // 画面を開いた初回以外で、かつ「対応中」への変化があった場合のみ音を鳴らす if (!isFirstLoad && triggerChime) { playChime(); } isFirstLoad = false; renderData(); }); }
